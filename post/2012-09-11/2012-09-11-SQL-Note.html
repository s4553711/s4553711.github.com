    Cretaing a temp table with 3 columns.

    CREATE TABLE #TEMP
(
    COL1 INT,
    COL2 VARCHAR(30),
    COL3 DATETIME DEFAULT GETDATE()
)
GO

    Getting value from temp table

    SELECT * FROM #TEMP
DROP TABLE #TEMP

    List all temp table

    SELECT name, create_date FROM TempDB.sys.tables WHERE name LIKE '#%'
